\textbf{Контрольная работа.} 
В каждом задании определите, какие утверждения истинны.
\begin{multicols}{2}
\bigskip

\textbf{1)} 
$A = \{1, \{1\}, \{1,2\}, \{\{1\}\}\}$\\
$1 \in A$\\
$\{1\} \in A$\\
$\{1,2\} \in A$\\
$2 \in A$\\
$\{\{1\}\} \in A$\\

\bigskip

\textbf{2)} 
$B = \{\varnothing, \{\varnothing\}, 0, \{0\}\}$\\
$\varnothing \in B$\\
$\{\varnothing\} \in B$\\
$0 \in B$\\
$\{0\} \in B$\\
$\{\{0\}\} \in B$\\

\bigskip

\textbf{3)} 
$K = \{\, 1,\ \{1,\varnothing\},\ \{\{1,\varnothing\}\},\ \{\{\{1\}\}\} \,\}$\\
$1 \in K$\\
$\{1\} \in K$\\
$\{1,\varnothing\} \in K$\\
$\{\{1,\varnothing\}\} \in K$\\
$\{\{\{1\}\}\} \in K$\\
$\{\{\{1,\varnothing\}\}\} \in K$\\

\bigskip

\textbf{4)} 
$C = \{a, \{a\}, \{a,b\}, b\}$\\
$a \in C$\\
$\{a\} \in C$\\
$\{b\} \in C$\\
$b \in C$\\
$\{a,b\} \in C$\\

\bigskip

\textbf{5)} 
$D = \{\{1\}, \{2\}, \{1,\{2\}\}, 2\}$\\
$1 \in D$\\
$\{1\} \in D$\\
$2 \in D$\\
$\{2\} \in D$\\
$\{1,\{2\}\} \in D$\\

\bigskip

\textbf{6)} 
$E = \{\varnothing, a, \{a,\varnothing\}, \{\{a\}\}\}$\\
$\varnothing \in E$\\
$a \in E$\\
$\{a\} \in E$\\
$\{a,\varnothing\} \in E$\\
$\{\{a\}\} \in E$\\

\bigskip

\textbf{7)} 
$L = \{\, \varnothing,\ \{\varnothing,0\},\ \{\{\varnothing,0\}\},\ 0,\ \{0,\{\varnothing\}\} \,\}$\\
$\varnothing \in L$\\
$\{\varnothing\} \in L$\\
$0 \in L$\\
$\{0\} \in L$\\
$\{\varnothing,0\} \in L$\\
$\{\{\varnothing,0\}\} \in L$\\
$\{0,\{\varnothing\}\} \in L$\\

\bigskip

\textbf{8)} 
$F = \{x, \{x\}, \{x,\{x\}\}, \{\{x\}\}\}$\\
$x \in F$\\
$\{x\} \in F$\\
$\{\{x\}\} \in F$\\
$\{x,\{x\}\} \in F$\\
$\{\{\{x\}\}\} \in F$\\

\bigskip

\textbf{9)} 
$G = \{0, 1, \{0,1\}, \{\{0,1\}\}\}$\\
$0 \in G$\\
$\{0\} \in G$\\
$\{0,1\} \in G$\\
$1 \in G$\\
$\{\{0,1\}\} \in G$\\

\bigskip

\textbf{10)} 
$H = \{\varnothing, \{1\}, \{1,\varnothing\}, \{\varnothing\}\}$\\
$\varnothing \in H$\\
$\{\varnothing\} \in H$\\
$1 \in H$\\
$\{1\} \in H$\\
$\{1,\varnothing\} \in H$\\
\end{multicols}
