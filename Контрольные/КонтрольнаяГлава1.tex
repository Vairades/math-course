\section{Итоговая контрольная работа по главе «Теория множеств» (45 минут)}

\textbf{1. (Определения, 6 баллов)}\\
Сформулируйте \textbf{точно}:
\begin{itemize}
\item определение подмножества,
\item определение декартова произведения,
\item определение пустого множества.
\end{itemize}

\medskip
\textbf{2. (Принадлежность и включение, 6 баллов)}\\
Пусть $S=\{\{1\},1,\{1,2\}\}$.  
Проверьте:
\[
1\in S,\quad
\{1\}\subset S,\quad
\{1,2\}\in S,\quad
\{2\}\in S.
\]

Найдите все подмножества множества всех подмножеств $T=\{0,\{0\}\}$.

\medskip
\textbf{3. (Операции, 8 баллов)}\\
Пусть $P=\{a,b,c\}$, $Q=\{b,c,d\}$. Найдите:
\[
P\cup Q,\quad
P\cap Q,\quad
P\setminus Q,\quad
Q\setminus P.
\]

Пусть $X=\{0,1\}$, $Y=\{5\}$.  
Найдите $(X\cup Y)\times(X\cap Y)$.

\medskip
\textbf{4. (Доказательство, 10 баллов)}\\
Докажите:
\[
A=B \Longleftrightarrow (A\subset B \text{ и } B\subset A).
\]

\medskip
\textbf{5. (Логика и кванторы, 7 баллов)}\\
Запишите формально и затем запишите отрицание утверждения:
\begin{quote}
«Для любого действительного числа $x>0$ найдётся число $y$, такое что $y^2>x$».
\end{quote}

\medskip
\textbf{6. (Доказательство от противного, 8 баллов)}\\
Докажите, что
\[
[0,1]\cap[1,2]=\{1\}.
\]

\section*{Экзаменационная работа по теме «Операции с множествами»}

\textbf{Время выполнения:} 90 минут

\bigskip

\subsection*{Часть A (базовый уровень)}

\begin{enumerate}
\item Задайте через свойства:
\begin{enumerate}
\item множество нечётных натуральных чисел;
\item множество квадратов натуральных чисел.
\end{enumerate}

\item Найдите:
\[
\{1,2,3\}\cap \{2,3,4\}, 
\quad
\{1,2,3\}\setminus \{2,3,4\}.
\]

\item Выпишите все элементы множества:
\[
\{a,b\}\times \{1,2,3\}.
\]
\end{enumerate}

\bigskip

\subsection*{Часть B (понимание)}

\begin{enumerate}
\setcounter{enumi}{3}

\item Докажите или опровергните утверждение:
\[
A\subset B \Rightarrow A\cap B = A.
\]

\item Приведите контрпример к утверждению:
\[
(A\setminus B)\cup B = A.
\]

\item Докажите дистрибутивный закон:
\[
A\cap (B\cup C) = (A\cap B)\cup (A\cap C).
\]
\end{enumerate}

\bigskip

\subsection*{Часть C (повышенный уровень)}

\begin{enumerate}
\setcounter{enumi}{6}

\item Пусть
\[
A=\{1,2\},\quad B=\{a,b\},\quad C=\{1\}.
\]
Найдите:
\[
(A\times B)\cap (C\times B).
\]

\item Докажите:
\[
(A\cap B)\times C = (A\times C)\cap (B\times C).
\]

\item Приведите пример множеств $A$ и $B$, для которых
\[
A\times B = B\times A.
\]
\end{enumerate}

\bigskip

\end{enumerate})
