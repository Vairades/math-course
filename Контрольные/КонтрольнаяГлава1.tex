\section{Итоговая контрольная работа по главе «Теория множеств» (45 минут)}

\textbf{1. (Определения, 6 баллов)}\\
Сформулируйте \textbf{точно}:
\begin{itemize}
\item определение подмножества,
\item определение декартова произведения,
\item определение пустого множества.
\end{itemize}

\medskip
\textbf{2. (Принадлежность и включение, 6 баллов)}\\
Пусть $S=\{\{1\},1,\{1,2\}\}$.  
Проверьте:
\[
1\in S,\quad
\{1\}\subset S,\quad
\{1,2\}\in S,\quad
\{2\}\in S.
\]

Найдите все подмножества множества всех подмножеств $T=\{0,\{0\}\}$.

\medskip
\textbf{3. (Операции, 8 баллов)}\\
Пусть $P=\{a,b,c\}$, $Q=\{b,c,d\}$. Найдите:
\[
P\cup Q,\quad
P\cap Q,\quad
P\setminus Q,\quad
Q\setminus P.
\]

Пусть $X=\{0,1\}$, $Y=\{5\}$.  
Найдите $(X\cup Y)\times(X\cap Y)$.

\medskip
\textbf{4. (Доказательство, 10 баллов)}\\
Докажите:
\[
A=B \Longleftrightarrow (A\subset B \text{ и } B\subset A).
\]

\medskip
\textbf{5. (Логика и кванторы, 7 баллов)}\\
Запишите формально и затем запишите отрицание утверждения:
\begin{quote}
«Для любого действительного числа $x>0$ найдётся число $y$, такое что $y^2>x$».
\end{quote}

\medskip
\textbf{6. (Доказательство от противного, 8 баллов)}\\
Докажите, что
\[
[0,1]\cap[1,2]=\{1\}.
\]
